import numpy as np
import bpy


#set scene
scn = bpy.context.scene
scn.frame_start = 11
scn.frame_end = 200

#define origin (used later)
origin = (0,0,0)
 
# Create ico sphere
bpy.ops.mesh.primitive_ico_sphere_add(location=origin)
ob = bpy.context.object


#load beats array
beats = np.load('C:/Users/Admin/Desktop/Project2/TestFiles/outfile.npy')

for beat in beats:
    bpy.context.scene.frame_set(frame = beat//1)
    bpy.ops.object.randomize_transform()
    bpy.ops.anim.keyframe_insert()
    

#Play animation
bpy.ops.screen.animation_play(reverse=False, sync=False)
